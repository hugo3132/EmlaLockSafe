77 bool checkTime() const {
78 // compute the number of seconds of this day
79 time_t currentTime = time(NULL);
80 tm tmBuf;
81 localtime_r(&currentTime, &tmBuf);
82 uint32_t currentSeconds = tmBuf.tm_hour * 3600 + tmBuf.tm_min * 60 + tmBuf.tm_sec;
83
84 if(startTime == endTime) {
85 return true;
86 }
87 else if (endTime > startTime) {
88 return (currentSeconds >= startTime) && (currentSeconds <= endTime);
89 }
90 else {
91 // Midnight overflow
92 return (currentSeconds >= endTime) && (currentSeconds <= startTime);
93 }
